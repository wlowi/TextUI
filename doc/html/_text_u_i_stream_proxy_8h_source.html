58/* How a uint8_t is encoded for transfer.
59 *
60 * A value in range 0 .. 63 is encoded as a single character with offset 48 ('0').
61 * The result is a printable character in range '0' to 'o'
62 *
63 * A value in range 64 .. 255 is encoded as two characters where the first
64 * character represents the high nibble and the second character represents
65 * the low nibble. The offset is 32 (' ', space character).
66 * The characters are two printable characters in range ' ' to '/'
67 */
68const uint8_t SINGLECHAR_OFFSET = 48;
69const uint8_t TWOCHAR_OFFSET = 32;
