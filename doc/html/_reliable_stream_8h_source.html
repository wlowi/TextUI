148 inline uint8_t outstanding() {
149 if( sendSeq >= ackdSeq) {
150 return sendSeq - ackdSeq;
151 } else {
152 return SEQUENCE_COUNT - (ackdSeq - sendSeq);
153 }
154 }

156 inline bool validAck( uint8_t ack) {
157
158 uint8_t miss;
159
160 if( ack >= ackdSeq) {
161 miss = ack - ackdSeq;
162 } else {
163 miss = SEQUENCE_COUNT - (ackdSeq - ack);
164 }
165
166 return (miss <= outstanding());
167 }
